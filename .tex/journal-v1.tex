\newcommand{\iarticle}[4]{
	% argv: author, title, tagline, noteline
	\clearpage
	\stepcounter{chapter}
	\phantomsection\addcontentsline{toc}{chapter}{\numberline{\thechapter}#2}
	\noindent\parbox{\linewidth}{
		{\small\sffamily\bfseries#3}\par\vskip 10pt
			{\LARGE\rmfamily\bfseries#2}\par\vskip 10pt
			{\footnotesize\rmfamily\mdseries#4}\par\vskip 1pt
		\rule{\linewidth}{1.5pt}
	}\par
    \vskip 15pt
}
